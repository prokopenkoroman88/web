Step 1: Declare the command to get product id and sku
Шаг 1. Объявите команду для получения идентификатора продукта и артикула
You will use a block class of the module Mageplaza_HelloWorld, then possibly inject the object of \Magento\Catalog\Model\ProductRepository class in the constructor of the module’s block class.
Вы будете использовать блочный класс модуля Mageplaza_HelloWorld, а затем, возможно, внедрить объект класса \Magento\Catalog\Model\ProductRepository в конструктор блочного класса модуля.

app/code/Mageplaza/HelloWorld/Block/HelloWorld.php

<?php
namespace Mageplaza\HelloWorld\Block;
class HelloWorld extends \Magento\Framework\View\Element\Template
{	
	protected $_productRepository;
		
	public function __construct(
		\Magento\Backend\Block\Template\Context $context,		
		\Magento\Catalog\Model\ProductRepository $productRepository,
		array $data = []
	)
	{
		$this->_productRepository = $productRepository;
		parent::__construct($context, $data);
	}
	
	public function getProductById($id)
	{
		return $this->_productRepository->getById($id);
	}
	
	public function getProductBySku($sku)
	{
		return $this->_productRepository->get($sku);
	}
}
?>
